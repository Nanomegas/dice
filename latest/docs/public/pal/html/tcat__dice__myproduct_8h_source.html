00037 //  Change this to customize the software for your DICE(tm) product line:

00039 //  You have to change all defines to make your driver coexist with other 
00040 //  customizations.
00041 //
00042 //  kTCAT_DICE_VENDOR

00044 #define kTCAT_DICE_VENDOR                   TC Applied Technologies

00046 //  kTCAT_DICE_VENDOR_SHORT 
00047 //  your company short form used for device matching. 
00048 //  Textual match for firmware config rom. This cannot contain any 
00049 //  non-alphanumeric character! Use underscores instead!
00050 #define kTCAT_DICE_VENDOR_SHORT             TCAT
00051 
00052 //  kTCAT_DICE_VENDOR_ID

00054 #define kTCAT_DICE_VENDOR_ID                0x166
00055 
00056 //  kTCAT_DICE_USR 
00057 //  will be displayed to the end user in any audio or midi driver selection 
00058 //  dialogs.
00059 //  This does not necessarily reflect your device name - since more of your 
00060 //  devices can be supported by that software. If you plan to only support one
00061 //  device with this software give it a name related to your device.
00062 //  Make sure this name does not exceed 32 character including eventual 
00063 //  enumeration additions made by the OS. If you also enable additions like
00064 //  Audio, MIDI, ASIO etc. in tcat_dice_macros.h all needs to fit.
00065 //  This is an OS limitation.
00066 #define kTCAT_DICE_USR                      TCAT Dice EVM Platform
00067 
00068 //  kTCAT_DICE_DRV 
00069 //  will be used to generate filenames and internally used symbolic 
00070 //  links.
00071 //  This does not necessarily reflect your device name - since more of your 
00072 //  devices can be supported by that software. If you plan to only support one
00073 //  device with this software give it a name related to your device.
00074 #define kTCAT_DICE_DRV                      TcatDice
00075 
00076 //  kTCAT_DICE_GUID 
00077 //  is a GUID that you pick to be distinctive from other DICE products that 
00078 //  might be installed in the same system. 
00079 //  (Unlike the static firmware magic number that is just an ID to find some 
00080 //  information in the binary file.)
00081 #define kTCAT_DICE_GUID                     {0x7ca7d1ce, 0x6cd2, 0x4e81, {0xb9, 0xf2, 0x71, 0x2d, 0x52, 0x15, 0x58, 0x50}}
