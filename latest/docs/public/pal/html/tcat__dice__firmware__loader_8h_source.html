00099 //Valid Test Command ID's, Parameter in FL_TEST_ACTION_PARAM, return in FL_TEST_RTN
00100 #define FL_TEST_CMD_POKE            1       //Write value to location, lvalue0=32 bit address, lvalue1=32 bit data
00101 #define FL_TEST_CMD_PEEK            2       //Read value from location, lvalue0=32 bit address, return data[0] = 32 bit value
00102 #define FL_TEST_CMD_GET_AVS_CNT     3       //Return AVS counters in data[NUM_AVS_EVENTS].
00103 #define FL_TEST_CMD_CLR_AVS_CNT     4       //Clear AVS counters
00104 #define FL_TEST_CMD_SET_MODE        5       //Set the EVM board mode equal to lvalue0, invoked by myModeSetMode
00105 #define FL_TEST_CMD_SET_MIDIBP      6       //Set the EVM (or customer product) to bypass MIDI if lvalue!=0, invoked by myModeBypassMidi
00106 #define FL_TEST_CMD_GET_AVS_PHASE   7       //Return the relative phase between PLL and Avs receiver, lvalue=0/1 (for Avs1/Avs2), return
00107                                             //  data[0]=signed phase offset in steps of 0.1us.
00108                                             //  data[1]=signed phase offset in %.
00109 #define FL_TEST_CMD_ENABLE_BNC_SYNC 8       //WCKO is programmed to pulse on AVS errors
00110 #define FL_TEST_CMD_PULSE_BNC_SYNC  9       //Send out pulse on WCKO
00111 #define FL_TEST_CMD_ENTER_TESTMODE  10      //Enter a firmware test mode
00112 
00113 #define FL_TEST_CMD_EMUL_SLOW_CMD   127     //This command does nothing but busy waits for the number of ms in lvalue0 before clearing the exec. flag.
